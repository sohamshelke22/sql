create or replace procedure proc_grade (
    prno studmarks.rno%TYPE
)
IS
    pname studmarks.name%type;
    pmarks studmarks.totalmarks%type;
    pclass varchar(100);
begin
    select name,totalmarks into pname,pmarks
    from studmarks where rno=prno;

    if pmarks between 990 and 1500 then
        pclass:='distinction';
    elsif pmarks between 900 and 989 then
        pclass:='first class';
    elsif pmarks between 825 and 899 then
        pclass:='higher second class';
    else
        pclass:='no class';
    end if;

    insert into result values(prno,pname,pclass);
end;
/
DECLARE
    prno number:=&enter_roll_no;
begin
    proc_grade(prno);
end;
/
